\begin{frame}{Architecture Overview}
\footnotesize
\begin{comparison}{Shared Infrastructure}{Class Hierarchy}
  \begin{description}[leftmargin=0pt]
    \item[\pkg{rossbeamerbase.sty}] ~\\
      Core beamer engine: colour derivation, font engine,
      progress bar, footer/logo system, all shared environments
      and commands. Loaded automatically by every beamer class.

    \item[\pkg{rosspalettes.sty}] ~\\
      Standalone palette system for non-beamer classes.
      Defines \texttt{rpPrimary}, \texttt{rpAccent}, etc.
      Used by manuscript, thesis, CV, abstract, letter, poster.

    \item[\pkg{rossstats.sty}] Statistical formatting helpers.
    \item[\pkg{rosstables.sty}] Table theming (any doc class).
    \item[\pkg{rossfigures.sty}] Figure panel layouts.
    \item[\pkg{rosscode.sty}] Code listing with minted.
  \end{description}
\comparisondivide
  \textbf{Beamer classes} (all load \pkg{rossbeamerbase}):
  \begin{itemize}\setlength\itemsep{1pt}
    \item \cls{rossbeamerscientific} --- research talks
    \item \cls{rossbeamercorporate} --- business decks
    \item \cls{rossbeamerpitch} --- startup / grant pitches
    \item \cls{rossbeamerminimal} --- Tufte-style sparse
    \item \cls{rossbeamermaths} --- theorems, proofs
    \item \cls{rossbeamerneuroimaging} --- fMRI / PET
    \item \cls{rossbeamerbiomarker} --- lab / assay
    \item \cls{rossbeamergenomics} --- genomics / variants
    \item \cls{rossbeamerclinical} --- clinical trials
    \item \cls{rossbeamerepi} --- epidemiology / DAGs
    \item \cls{rossbeamerpsych} --- neuropsych assessment
    \item \cls{rossbeamerteaching} --- lectures / workshops
    \item \cls{rossbeamerposter} --- conference posters
  \end{itemize}
  \vskip4pt
  \textbf{Document classes} (all load \pkg{rosspalettes}):
  \begin{itemize}\setlength\itemsep{1pt}
    \item \cls{rossmanuscript} \cls{rossthesis} \cls{rosscv}
    \item \cls{rossabstract} \cls{rossletter} \cls{rossposter}
  \end{itemize}
\end{comparison}
\end{frame}

\begin{frame}{Common Options}
\footnotesize
All beamer classes accept these options in \texttt{\textbackslash documentclass[...]\{class\}}:
\vskip6pt
\begin{tabular}{@{}l l p{7.5cm}@{}}
  \rowcolor{rbPrimaryFaded}
  \thead{Option} & \thead{Default} & \thead{Description} \\
  \opt{palette=\textit{name}} & varies & Colour palette (see palette list) \\
  \rowcolor{rbPrimaryFaded}
  \opt{font=\textit{name}} & \texttt{fira} & Sans font: \texttt{fira}, \texttt{inter}, \texttt{sourcesans}, \texttt{libertinus}, \texttt{heros}, \texttt{georgia}, \texttt{system} \\
  \opt{monofont=\textit{name}} & Fira Code & Override monospace font name \\
  \rowcolor{rbPrimaryFaded}
  \opt{nobar} & off & Disable progress bar \\
  \opt{nologo} & off & Disable logo in footer \\
  \rowcolor{rbPrimaryFaded}
  \opt{darkmode} & off & Dark background (corporate, pitch only) \\
  \opt{chalkboard} & off & Green chalkboard mode (maths only) \\
  \rowcolor{rbPrimaryFaded}
  \opt{stats} & off & Load \pkg{rossstats.sty} (biomarker, clinical, psych) \\
  \opt{loadstats} & off & Load \pkg{rossstats.sty} (epi) \\
\end{tabular}
\vskip8pt
\begin{twocol}
  \textbf{Preamble configuration:}
  \begin{itemize}\setlength\itemsep{1pt}
    \item \cmd{slidelogo[height=6mm]\{file.png\}}
    \item \cmd{footertext\{Company Name\}}
    \item \cmd{slidenote\{Confidential\}}
    \item \cmd{logoposition\{bottomleft\}} --- also \texttt{bottomright}, \texttt{topleft}, \texttt{topright}
    \item \cmd{customfontname\{FontName\}} --- for \texttt{font=system}
  \end{itemize}
\colbreak
  \textbf{Runtime commands:}
  \begin{itemize}\setlength\itemsep{1pt}
    \item \cmd{setpalette\{name\}} --- switch palette mid-document
    \item \cmd{enabledraft[text]} --- add watermark
    \item \cmd{enablesectiondividers} --- auto dark dividers
    \item \cmd{enablefooter} --- activate footer (corporate)
    \item \cmd{conference\{name\}} --- subtitle on title page (scientific, clinical, etc.)
  \end{itemize}
\end{twocol}
\end{frame}

\begin{frame}{Shared Environments \normalfont\small(rossbeamerbase.sty)}

      \texttt{\textbackslash begin\{comparison\}\{Left\}\{Right\}...}\\
      \texttt{\textbackslash comparisondivide ...}
    \item[\env{twocol}] Quick two-column.\\
      \texttt{\textbackslash begin\{twocol\}[0.48]...}\\
      \texttt{\textbackslash colbreak[0.48]...}
    \item[\env{figurepanel}] Centred figure with caption text.
    \item[\env{referenceslide}] Bibliography frame (scientific classes).
  \end{description}

  \begin{description}[leftmargin=0pt]
    \item[\env{card}] White box, primary left border.
    \item[\env{accentcard}] Tinted box, accent left border.
    \item[\env{highlight}] Accent-tinted highlight box.
    \item[\env{notebox}] Muted note box.
    \item[\env{warningbox}] Amber warning box.
    \item[\env{successbox}] Green success box.
    \item[\env{infobox}] Blue info box.
    \item[\env{eqbox}] Boxed equation (scientific, maths).
  \end{description}
  \vskip4pt
  All boxes accept tcolorbox options and an optional \texttt{title=\{...\}} key.
  \vskip4pt
  \textbf{Standalone commands:}
  \begin{itemize}\setlength\itemsep{0pt}
    \item \cmd{sectiondivider} --- manual dark section slide
    \item \cmd{conclusionslide[title]\{subtitle\}}
    \item \cmd{titleonlyframe[subtitle]\{Title\}}
    \item \cmd{quoteslide\{quote\}\{attribution\}}
    \item \cmd{fullbleedimage[overlay]\{file\}}
  \end{itemize}
\end{comparison}
\end{frame}

\begin{frame}{Standalone Packages}
\footnotesize
These can be loaded in \textbf{any} \LaTeX\ document with \texttt{\textbackslash usepackage\{...\}}.
They are independent of the beamer and document classes.
\vskip6pt
\begin{twocol}
  \textbf{\pkg{rossstats.sty}} --- Statistical notation
  \begin{itemize}\setlength\itemsep{0pt}
    \item \cmd{pval\{.003\}} --- auto $< .001$ for tiny values
    \item \cmd{ci\{est\}\{lo\}\{hi\}}, \cmd{CI\{level\}\{est\}\{lo\}\{hi\}}
    \item \cmd{effectsize\{d\}\{0.72\}}
    \item \cmd{chisq\{val\}}, \cmd{fstat\{df1\}\{df2\}\{val\}}
    \item \cmd{tstat\{df\}\{val\}}, \cmd{betacoef\{val\}}
    \item \cmd{OR\{est\}\{lo\}\{hi\}}, \cmd{HR\{...\}}, \cmd{RR\{...\}}
    \item \cmd{NNT\{n\}}, \cmd{NNH\{n\}}, \cmd{ARR\{val\}}
    \item \cmd{sensspec\{sens\}\{spec\}}, \cmd{ppvnpv\{ppv\}\{npv\}}
    \item \cmd{rci\{val\}}, \cmd{prevalence\{pct\}}, \cmd{incidence\{...\}}
    \item \cmd{rateratio\{...\}}, \cmd{oddsratio\{...\}}, \cmd{attributablerisk\{...\}}
    \item \cmd{sigmarkers}, \cmd{figurenote\{...\}}, \cmd{tablenote\{...\}}
  \end{itemize}
  \vskip6pt

  \textbf{\pkg{rosstables.sty}} --- Table theming
  \begin{itemize}\setlength\itemsep{0pt}
    \item \cmd{thead\{text\}} --- bold coloured header cell
    \item \cmd{shaderow} --- alternating row shading
    \item \cmd{tablehead\{text\}} --- full-width header row
    \item \env{rosstable}\{caption\} --- themed table environment
    \item \cmd{tablenotesig} --- significance markers below table
  \end{itemize}

\colbreak
  \textbf{\pkg{rossfigures.sty}} --- Figure layouts
  \begin{itemize}\setlength\itemsep{0pt}
    \item \env{figurepanel}[caption] --- centred figure with caption
    \item \cmd{panel[width]\{file\}\{caption\}} --- sub-panel
    \item \cmd{figuregrid\{cols\}\{items\}} --- auto-grid of panels
    \item \cmd{fullwidthfigure\{file\}\{caption\}} --- span text width
    \item \cmd{sidebyside\{fileL\}\{capL\}\{fileR\}\{capR\}}
    \item \cmd{source\{text\}} --- italicised source line
  \end{itemize}
  \vskip6pt

  \textbf{\pkg{rosscode.sty}} --- Code listings (minted)
  \begin{itemize}\setlength\itemsep{0pt}
    \item \cmd{codeinline[lang]\{code\}} --- inline code snippet
    \item \cmd{codefile[lang]\{path\}} --- import external file
    \item Also provides the \env{code} environment (via minted)
      with palette-aware styling.
  \end{itemize}
