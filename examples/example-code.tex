\begin{codechunk}[R]
library(tidyverse)
library(lavaan)

# Load and prepare data
sleep <- read_csv("sleep_cognition.csv") %>%
  mutate(
    sws_pct  = sws_minutes / tst_minutes * 100,
    dass_dep = rowSums(select(., starts_with("dass_d"))),
    dass_anx = rowSums(select(., starts_with("dass_a")))
  )

# Parallel mediation model
model <- "
  delayed_memory ~ c*sws_pct + b1*dass_dep + b2*dass_anx
  dass_dep       ~ a1*sws_pct
  dass_anx       ~ a2*sws_pct
  indirect_dep := a1 * b1
  indirect_anx := a2 * b2
  total          := c + a1*b1 + a2*b2
"
fit <- sem(model, data = sleep, se = "bootstrap", bootstrap = 5000)
summary(fit, ci = TRUE, standardized = TRUE)
\end{codechunk}

\section*{2. Python code chunk}

\begin{codechunk}[Python]
import pandas as pd
import pingouin as pg

df = pd.read_csv("sleep_cognition.csv")

# Partial correlation: SWS% and delayed memory, controlling for age
partial = pg.partial_corr(
    data=df,
    x="sws_pct",
    y="delayed_memory",
    covar=["age", "education"],
    method="pearson",
)
print(partial.round(3))
\end{codechunk}

\section*{4. Mixed narrative}

After fitting the mediation model in R, the indirect effect through
depressive symptoms was $\beta = -0.18$, 95\% CI [$-0.34$, $-0.08$],
$p = .002$. The Python partial-correlation analysis confirmed the
zero-order association ($r = -.24$, $p = .001$) after controlling for
age and education.
