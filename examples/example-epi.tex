% --- Finding ---
\begin{frame}{Conclusions}
  \begin{findingbox}
    Occupational dust exposure was associated with a 7.7-fold increased
    odds of chronic bronchitis after adjustment for smoking and SES.
  \end{findingbox}
  \vskip6pt
  \begin{implicationbox}[Public Health Implication]
    Enhanced workplace ventilation standards in textile factories
    could substantially reduce respiratory disease burden.
  \end{implicationbox}
